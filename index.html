Fonctions essentielles et exemples pratiques

setup()

Initialisation exécutée une fois au démarrage.

void setup() { ... } 
void setup() {
  pinMode(13, OUTPUT);
  Serial.begin(9600);
}

loop()

Boucle principale exécutée continuellement.

void loop() { ... } 
void loop() {
  digitalWrite(13, HIGH);
  delay(1000);
  digitalWrite(13, LOW);
  delay(1000);
}

pinMode()

Configure une broche en entrée ou sortie.

pinMode(pin, mode) 
pinMode(13, OUTPUT);  // Sortie
pinMode(2, INPUT);     // Entrée
pinMode(3, INPUT_PULLUP); // Entrée avec pull-up

digitalWrite()

Écrit une valeur HIGH ou LOW sur une broche.

digitalWrite(pin, value) 
digitalWrite(13, HIGH); // Allume
digitalWrite(13, LOW);  // Éteint

digitalRead()

Lit l'état d'une broche d'entrée.

digitalRead(pin) 
int state = digitalRead(2);
if (state == HIGH) {
  // Bouton appuyé
}

analogWrite()

Génère un signal PWM (0-255).

analogWrite(pin, value) 
analogWrite(9, 128); // PWM 50%
analogWrite(10, 255); // PWM 100%

analogRead()

Lit une valeur analogique (0-1023).

analogRead(pin) 
int sensor = analogRead(A0);
float voltage = sensor * (5.0 / 1023.0);

delay()

Pause en millisecondes.

delay(ms) 
delay(1000);  // 1 seconde
delay(500);   // 0.5 seconde
